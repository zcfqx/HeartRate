# 心率监控软件数据库设计文档

## 1. 数据库概述

### 1.1 技术选型
- **数据库引擎**：SQLite
- **ORM框架**：Dapper
- **数据库版本**：SQLite 3.x

### 1.2 设计原则
- **轻量级**：SQLite适合桌面应用，无需额外服务
- **高性能**：合理设计索引，优化查询性能
- **可维护**：清晰的表结构和字段命名
- **可扩展**：预留扩展字段

## 3. 表结构设计

### 3.1 心率记录表 (HeartRateRecord)

```sql
-- 心率记录表
CREATE TABLE HeartRateRecord (
    Id              INTEGER PRIMARY KEY AUTOINCREMENT,
    Timestamp       DATETIME NOT NULL,
    HeartRate       INTEGER NOT NULL,
    DeviceId        TEXT NOT NULL,
    BatteryLevel    INTEGER,
    SensorContact   BOOLEAN,
    EnergyExpended  INTEGER,
    IsAbnormal      BOOLEAN DEFAULT FALSE,
    ZoneType        INTEGER DEFAULT 0,
    CreatedAt       DATETIME DEFAULT CURRENT_TIMESTAMP,
    
    CONSTRAINT FK_HeartRateRecord_Device 
        FOREIGN KEY (DeviceId) REFERENCES DeviceInfo(Id),
    CONSTRAINT CK_HeartRate_Valid 
        CHECK (HeartRate >= 0 AND HeartRate <= 255),
    CONSTRAINT CK_Battery_Valid 
        CHECK (BatteryLevel IS NULL OR (BatteryLevel >= 0 AND BatteryLevel <= 100))
);

-- 索引
CREATE INDEX idx_heart_rate_timestamp ON HeartRateRecord(Timestamp);
CREATE INDEX idx_heart_rate_device ON HeartRateRecord(DeviceId);
CREATE INDEX idx_heart_rate_device_timestamp ON HeartRateRecord(DeviceId, Timestamp);
CREATE INDEX idx_heart_rate_created ON HeartRateRecord(CreatedAt);
```

#### 字段说明

| 字段 | 类型 | 必填 | 说明 |
|------|------|------|------|
| Id | INTEGER | 是 | 自增主键 |
| Timestamp | DATETIME | 是 | 心率测量时间 |
| HeartRate | INTEGER | 是 | 心率值（0-255 BPM） |
| DeviceId | TEXT | 是 | 设备ID，外键关联DeviceInfo |
| BatteryLevel | INTEGER | 否 | 电量百分比（0-100） |
| SensorContact | BOOLEAN | 否 | 传感器接触状态 |
| EnergyExpended | INTEGER | 否 | 能量消耗（kJ） |
| IsAbnormal | BOOLEAN | 是 | 是否异常（超出阈值） |
| ZoneType | INTEGER | 是 | 心率区间类型（0-4） |
| CreatedAt | DATETIME | 是 | 记录创建时间 |

## 5. 数据库初始化

### 5.1 DatabaseInitializer

```csharp
namespace HeartRateMonitor.Data.Database
{
    /// <summary>
    /// 数据库初始化器
    /// </summary>
    public class DatabaseInitializer
    {
        private readonly string _connectionString;
        
        public DatabaseInitializer(string dbPath)
        {
            _connectionString = $"Data Source={dbPath}";
        }
        
        /// <summary>
        /// 初始化数据库
        /// </summary>
        public async Task InitializeAsync()
        {
            using var connection = new SqliteConnection(_connectionString);
            await connection.OpenAsync();
            
            // 创建表
            await CreateTablesAsync(connection);
            
            // 插入默认配置
            await InsertDefaultSettingsAsync(connection);
            
            // 执行迁移
            await MigrateAsync(connection);
        }
        
        private async Task CreateTablesAsync(SqliteConnection connection)
        {
            var sql = @"
                -- 设备信息表
                CREATE TABLE IF NOT EXISTS DeviceInfo (
                    Id              TEXT PRIMARY KEY,
                    Name            TEXT NOT NULL,
                    MacAddress      TEXT,
                    DeviceType      INTEGER DEFAULT 0,
                    FirmwareVersion TEXT,
                    LastConnected   DATETIME,
                    TotalConnections INTEGER DEFAULT 0,
                    TotalDuration   INTEGER DEFAULT 0,
                    IsAutoConnect   BOOLEAN DEFAULT TRUE,
                    IsFavorite      BOOLEAN DEFAULT FALSE,
                    Notes           TEXT,
                    CreatedAt       DATETIME DEFAULT CURRENT_TIMESTAMP,
                    UpdatedAt       DATETIME DEFAULT CURRENT_TIMESTAMP
                );
                
                -- 心率记录表
                CREATE TABLE IF NOT EXISTS HeartRateRecord (
                    Id              INTEGER PRIMARY KEY AUTOINCREMENT,
                    Timestamp       DATETIME NOT NULL,
                    HeartRate       INTEGER NOT NULL,
                    DeviceId        TEXT NOT NULL,
                    BatteryLevel    INTEGER,
                    SensorContact   BOOLEAN,
                    EnergyExpended  INTEGER,
                    IsAbnormal      BOOLEAN DEFAULT FALSE,
                    ZoneType        INTEGER DEFAULT 0,
                    CreatedAt       DATETIME DEFAULT CURRENT_TIMESTAMP,
                    CONSTRAINT FK_HeartRateRecord_Device 
                        FOREIGN KEY (DeviceId) REFERENCES DeviceInfo(Id)
                );
                
                -- 配置表
                CREATE TABLE IF NOT EXISTS Settings (
                    Key         TEXT PRIMARY KEY,
                    Value       TEXT NOT NULL,
                    ValueType   INTEGER DEFAULT 0,
                    Description TEXT,
                    UpdatedAt   DATETIME DEFAULT CURRENT_TIMESTAMP
                );
                
                -- 创建索引
                CREATE INDEX IF NOT EXISTS idx_heart_rate_timestamp 
                    ON HeartRateRecord(Timestamp);
                CREATE INDEX IF NOT EXISTS idx_heart_rate_device 
                    ON HeartRateRecord(DeviceId);
                CREATE INDEX IF NOT EXISTS idx_heart_rate_device_timestamp 
                    ON HeartRateRecord(DeviceId, Timestamp);
                CREATE INDEX IF NOT EXISTS idx_device_last_connected 
                    ON DeviceInfo(LastConnected);
            ";
            
            using var command = new SqliteCommand(sql, connection);
            await command.ExecuteNonQueryAsync();
        }
        
        private async Task InsertDefaultSettingsAsync(SqliteConnection connection)
        {
            var settings = new Dictionary<string, (string Value, int Type, string Desc)>
            {
                ["App.Theme"] = ("Dark", 0, "主题类型"),
                ["App.Language"] = ("zh-CN", 0, "语言设置"),
                ["App.Opacity"] = ("0.9", 1, "窗口透明度"),
                ["App.TopMost"] = ("true", 2, "窗口置顶"),
                ["App.AutoStart"] = ("false", 2, "开机自启动"),
                ["HeartRate.UpperLimit"] = ("180", 1, "心率上限报警"),
                ["HeartRate.LowerLimit"] = ("40", 1, "心率下限报警"),
                ["HeartRate.AlertEnabled"] = ("true", 2, "启用心率报警"),
                ["Ble.AutoConnect"] = ("true", 2, "自动连接设备"),
                ["Ble.ScanTimeout"] = ("30", 1, "扫描超时"),
                ["Data.RetentionDays"] = ("30", 1, "数据保留天数")
            };
            
            foreach (var setting in settings)
            {
                var sql = @"
                    INSERT OR IGNORE INTO Settings (Key, Value, ValueType, Description)
                    VALUES (@Key, @Value, @Type, @Desc)";
                
                using var command = new SqliteCommand(sql, connection);
                command.Parameters.AddWithValue("@Key", setting.Key);
                command.Parameters.AddWithValue("@Value", setting.Value.Value);
                command.Parameters.AddWithValue("@Type", setting.Value.Type);
                command.Parameters.AddWithValue("@Desc", setting.Value.Desc);
                await command.ExecuteNonQueryAsync();
            }
        }
        
        private async Task MigrateAsync(SqliteConnection connection)
        {
            // 获取当前数据库版本
            var version = await GetDatabaseVersionAsync(connection);
            
            // 执行迁移脚本
            if (version < 1)
            {
                await MigrateToV1Async(connection);
            }
            if (version < 2)
            {
                await MigrateToV2Async(connection);
            }
            // ... 更多迁移
        }
        
        private async Task<int> GetDatabaseVersionAsync(SqliteConnection connection)
        {
            var sql = "PRAGMA user_version";
            using var command = new SqliteCommand(sql, connection);
            var result = await command.ExecuteScalarAsync();
            return Convert.ToInt32(result);
        }
        
        private async Task MigrateToV1Async(SqliteConnection connection)
        {
            // V1迁移脚本
            var sql = "PRAGMA user_version = 1";
            using var command = new SqliteCommand(sql, connection);
            await command.ExecuteNonQueryAsync();
        }
        
        private async Task MigrateToV2Async(SqliteConnection connection)
        {
            // V2迁移脚本
            var sql = @"
                ALTER TABLE DeviceInfo ADD COLUMN DeviceType INTEGER DEFAULT 0;
                ALTER TABLE DeviceInfo ADD COLUMN FirmwareVersion TEXT;
                PRAGMA user_version = 2";
            using var command = new SqliteCommand(sql, connection);
            await command.ExecuteNonQueryAsync();
        }
    }
}
```

```csharp
namespace HeartRateMonitor.Data.Database
{
    /// <summary>
    /// 数据库上下文
    /// </summary>
    public class DbContext : IDisposable
    {
        private readonly string _connectionString;
        private SqliteConnection _connection;
        
        public DbContext(string dbPath)
        {
            _connectionString = $"Data Source={dbPath}";
        }
        
        /// <summary>
        /// 获取数据库连接
        /// </summary>
        public async Task<SqliteConnection> GetConnectionAsync()
        {
            if (_connection == null)
            {
                _connection = new SqliteConnection(_connectionString);
                await _connection.OpenAsync();
                
                // 启用WAL模式提高并发性能
                using var command = new SqliteCommand("PRAGMA journal_mode=WAL", _connection);
                await command.ExecuteNonQueryAsync();
            }
            
            return _connection;
        }
        
        public void Dispose()
        {
            _connection?.Dispose();
        }
    }
}
```

### 6.2 HeartRateRepository

```csharp
namespace HeartRateMonitor.Data.Repositories
{
    /// <summary>
    /// 心率记录仓库实现
    /// </summary>
    public class HeartRateRepository : IHeartRateRepository
    {
        private readonly DbContext _dbContext;
        
        public HeartRateRepository(DbContext dbContext)
        {
            _dbContext = dbContext;
        }
        
        public async Task<int> AddAsync(HeartRateRecord record)
        {
            var connection = await _dbContext.GetConnectionAsync();
            
            var sql = @"
                INSERT INTO HeartRateRecord 
                    (Timestamp, HeartRate, DeviceId, BatteryLevel, SensorContact, 
                     EnergyExpended, IsAbnormal, ZoneType)
                VALUES 
                    (@Timestamp, @HeartRate, @DeviceId, @BatteryLevel, @SensorContact,
                     @EnergyExpended, @IsAbnormal, @ZoneType);
                SELECT last_insert_rowid();";
            
            using var command = new SqliteCommand(sql, connection);
            command.Parameters.AddWithValue("@Timestamp", record.Timestamp);
            command.Parameters.AddWithValue("@HeartRate", record.HeartRate);
            command.Parameters.AddWithValue("@DeviceId", record.DeviceId);
            command.Parameters.AddWithValue("@BatteryLevel", (object)record.BatteryLevel ?? DBNull.Value);
            command.Parameters.AddWithValue("@SensorContact", (object)record.SensorContact ?? DBNull.Value);
            command.Parameters.AddWithValue("@EnergyExpended", (object)record.EnergyExpended ?? DBNull.Value);
            command.Parameters.AddWithValue("@IsAbnormal", record.IsAbnormal);
            command.Parameters.AddWithValue("@ZoneType", record.ZoneType);
            
            var result = await command.ExecuteScalarAsync();
            return Convert.ToInt32(result);
        }
        
        public async Task AddRangeAsync(IEnumerable<HeartRateRecord> records)
        {
            var connection = await _dbContext.GetConnectionAsync();
            
            using var transaction = connection.BeginTransaction();
            
            var sql = @"
                INSERT INTO HeartRateRecord 
                    (Timestamp, HeartRate, DeviceId, BatteryLevel, SensorContact, 
                     EnergyExpended, IsAbnormal, ZoneType)
                VALUES 
                    (@Timestamp, @HeartRate, @DeviceId, @BatteryLevel, @SensorContact,
                     @EnergyExpended, @IsAbnormal, @ZoneType)";
            
            foreach (var record in records)
            {
                using var command = new SqliteCommand(sql, connection, transaction);
                command.Parameters.AddWithValue("@Timestamp", record.Timestamp);
                command.Parameters.AddWithValue("@HeartRate", record.HeartRate);
                command.Parameters.AddWithValue("@DeviceId", record.DeviceId);
                command.Parameters.AddWithValue("@BatteryLevel", (object)record.BatteryLevel ?? DBNull.Value);
                command.Parameters.AddWithValue("@SensorContact", (object)record.SensorContact ?? DBNull.Value);
                command.Parameters.AddWithValue("@EnergyExpended", (object)record.EnergyExpended ?? DBNull.Value);
                command.Parameters.AddWithValue("@IsAbnormal", record.IsAbnormal);
                command.Parameters.AddWithValue("@ZoneType", record.ZoneType);
                
                await command.ExecuteNonQueryAsync();
            }
            
            await transaction.CommitAsync();
        }
        
        public async Task<IEnumerable<HeartRateRecord>> GetByTimeRangeAsync(
            DateTime startTime, DateTime endTime, string deviceId = null)
        {
            var connection = await _dbContext.GetConnectionAsync();
            
            var sql = @"
                SELECT * FROM HeartRateRecord 
                WHERE Timestamp BETWEEN @StartTime AND @EndTime";
            
            if (!string.IsNullOrEmpty(deviceId))
            {
                sql += " AND DeviceId = @DeviceId";
            }
            
            sql += " ORDER BY Timestamp";
            
            using var command = new SqliteCommand(sql, connection);
            command.Parameters.AddWithValue("@StartTime", startTime);
            command.Parameters.AddWithValue("@EndTime", endTime);
            if (!string.IsNullOrEmpty(deviceId))
            {
                command.Parameters.AddWithValue("@DeviceId", deviceId);
            }
            
            var records = new List<HeartRateRecord>();
            using var reader = await command.ExecuteReaderAsync();
            
            while (await reader.ReadAsync())
            {
                records.Add(MapToEntity(reader));
            }
            
            return records;
        }
        
        public async Task DeleteBeforeDateAsync(DateTime beforeDate)
        {
            var connection = await _dbContext.GetConnectionAsync();
            
            var sql = "DELETE FROM HeartRateRecord WHERE Timestamp < @BeforeDate";
            
            using var command = new SqliteCommand(sql, connection);
            command.Parameters.AddWithValue("@BeforeDate", beforeDate);
            await command.ExecuteNonQueryAsync();
        }
        
        private HeartRateRecord MapToEntity(SqliteDataReader reader)
        {
            return new HeartRateRecord
            {
                Id = reader.GetInt32(reader.GetOrdinal("Id")),
                Timestamp = reader.GetDateTime(reader.GetOrdinal("Timestamp")),
                HeartRate = reader.GetInt32(reader.GetOrdinal("HeartRate")),
                DeviceId = reader.GetString(reader.GetOrdinal("DeviceId")),
                BatteryLevel = reader.IsDBNull(reader.GetOrdinal("BatteryLevel")) 
                    ? null : reader.GetInt32(reader.GetOrdinal("BatteryLevel")),
                SensorContact = reader.IsDBNull(reader.GetOrdinal("SensorContact")) 
                    ? null : reader.GetBoolean(reader.GetOrdinal("SensorContact")),
                EnergyExpended = reader.IsDBNull(reader.GetOrdinal("EnergyExpended")) 
                    ? null : reader.GetInt32(reader.GetOrdinal("EnergyExpended")),
                IsAbnormal = reader.GetBoolean(reader.GetOrdinal("IsAbnormal")),
                ZoneType = reader.GetInt32(reader.GetOrdinal("ZoneType")),
                CreatedAt = reader.GetDateTime(reader.GetOrdinal("CreatedAt"))
            };
        }
    }
}
```

## 7. 数据维护策略

### 7.1 数据清理策略

```csharp
public class DataMaintenanceService
{
    private readonly IHeartRateRepository _heartRateRepository;
    private readonly ISettingsService _settingsService;
    
    /// <summary>
    /// 清理旧数据
    /// </summary>
    public async Task CleanupOldDataAsync()
    {
        var retentionDays = _settingsService.GetValue<int>("Data.RetentionDays", 30);
```

```csharp
public class BatchInsertOptimizer
{
    private readonly List<HeartRateRecord> _buffer = new();
    private readonly int _batchSize = 100;
    private readonly TimeSpan _flushInterval = TimeSpan.FromSeconds(5);
    
    public void Add(HeartRateRecord record)
    {
        _buffer.Add(record);
        
        if (_buffer.Count >= _batchSize)
        {
            FlushAsync().ConfigureAwait(false);
        }
    }
    
    public async Task FlushAsync()
    {
        if (_buffer.Count == 0) return;
        
        await _repository.AddRangeAsync(_buffer);
        _buffer.Clear();
    }
}
```
