```csharp
[TestClass]
public class HeartRateParserTests
{
    private HeartRateParser _parser;
    
    [TestInitialize]
    public void Setup()
    {
        _parser = new HeartRateParser();
    }
    
    [TestMethod]
    public void Parse_UINT8Format_ReturnsCorrectHeartRate()
    {
        // Arrange
        var data = new byte[] { 0x00, 0x48 }; // Flags=0, HR=72
        
        // Act
        var result = _parser.Parse(data);
        
        // Assert
        Assert.AreEqual(72, result.HeartRate);
    }
    
    [TestMethod]
    public void Parse_UINT16Format_ReturnsCorrectHeartRate()
    {
        // Arrange
        var data = new byte[] { 0x01, 0x48, 0x00 }; // Flags=1, HR=72
        
        // Act
        var result = _parser.Parse(data);
        
        // Assert
        Assert.AreEqual(72, result.HeartRate);
    }
    
    [TestMethod]
    public void Parse_WithSensorContact_SetsContactStatus()
    {
        // Arrange
        var data = new byte[] { 0x06, 0x48 }; // Flags=6 (contact supported + detected)
        
        // Act
        var result = _parser.Parse(data);
        
        // Assert
        Assert.IsTrue(result.SensorContact);
    }
    
    [TestMethod]
    public void Parse_WithEnergyExpended_ParsesCorrectly()
    {
        // Arrange
        var data = new byte[] { 0x08, 0x50, 0x00, 0x01 }; // Flags=8, HR=80, Energy=256
        
        // Act
        var result = _parser.Parse(data);
        
        // Assert
        Assert.AreEqual(80, result.HeartRate);
        Assert.AreEqual(256, result.EnergyExpended);
    }
    
    [TestMethod]
    public void Parse_WithRRInterval_ParsesCorrectly()
    {
        // Arrange
        var data = new byte[] { 0x10, 0x50, 0x00, 0x01 }; // Flags=16, HR=80, RR=256
        
        // Act
        var result = _parser.Parse(data);
        
        // Assert
        Assert.AreEqual(80, result.HeartRate);
        Assert.IsNotNull(result.RRIntervals);
        Assert.AreEqual(1, result.RRIntervals.Count);
    }
    
    [TestMethod]
    [ExpectedException(typeof(ArgumentException))]
    public void Parse_NullData_ThrowsException()
    {
        // Act
        _parser.Parse(null);
    }
    
    [TestMethod]
    [ExpectedException(typeof(ArgumentException))]
    public void Parse_EmptyData_ThrowsException()
    {
        // Act
        _parser.Parse(Array.Empty<byte>());
    }
    
    [TestMethod]
    public void IsValidFormat_ValidData_ReturnsTrue()
    {
        // Arrange
        var data = new byte[] { 0x00, 0x48 };
        
        // Act
        var result = _parser.IsValidFormat(data);
        
        // Assert
        Assert.IsTrue(result);
    }
    
    [TestMethod]
    public void IsValidFormat_InvalidHeartRate_ReturnsFalse()
    {
        // Arrange
        var data = new byte[] { 0x00 }; // 缺少心率数据
        
        // Act
        var result = _parser.IsValidFormat(data);
        
        // Assert
        Assert.IsFalse(result);
    }
}
```

```csharp
[TestClass]
[TestCategory("Integration")]
public class DataStorageIntegrationTests
{
    private string _testDbPath;
    private DbContext _dbContext;
    private HeartRateRepository _repository;
    
    [TestInitialize]
    public async Task Setup()
    {
        _testDbPath = Path.Combine(Path.GetTempPath(), $"test_{Guid.NewGuid()}.db");
        _dbContext = new DbContext(_testDbPath);
        _repository = new HeartRateRepository(_dbContext);
        
        // 初始化测试数据库
        var initializer = new DatabaseInitializer(_testDbPath);
        await initializer.InitializeAsync();
    }
    
    [TestCleanup]
    public void Cleanup()
    {
        _dbContext.Dispose();
        if (File.Exists(_testDbPath))
        {
            File.Delete(_testDbPath);
        }
    }
    
    [TestMethod]
    public async Task AddAsync_SavesRecord()
    {
        // Arrange
        var record = new HeartRateRecord
        {
            Timestamp = DateTime.Now,
            HeartRate = 72,
            DeviceId = "test-device",
            IsAbnormal = false,
            ZoneType = 0
        };
        
        // Act
        var id = await _repository.AddAsync(record);
        
        // Assert
        Assert.IsTrue(id > 0);
        
        var saved = await _repository.GetByIdAsync(id);
        Assert.IsNotNull(saved);
        Assert.AreEqual(72, saved.HeartRate);
    }
    
    [TestMethod]
    public async Task GetByTimeRangeAsync_ReturnsCorrectRecords()
    {
        // Arrange
        var startTime = DateTime.Now.AddMinutes(-5);
        var endTime = DateTime.Now;
        
        await _repository.AddAsync(new HeartRateRecord
        {
            Timestamp = startTime.AddMinutes(1),
            HeartRate = 70,
            DeviceId = "test-device"
        });
        
        await _repository.AddAsync(new HeartRateRecord
        {
            Timestamp = startTime.AddMinutes(2),
            HeartRate = 75,
            DeviceId = "test-device"
        });
        
        // Act
        var records = await _repository.GetByTimeRangeAsync(startTime, endTime);
        
        // Assert
        Assert.AreEqual(2, records.Count());
    }
    
    [TestMethod]
    public async Task DeleteBeforeDateAsync_RemovesOldRecords()
    {
        // Arrange
        await _repository.AddAsync(new HeartRateRecord
        {
            Timestamp = DateTime.Now.AddDays(-31),
            HeartRate = 70,
            DeviceId = "test-device"
        });
        
        await _repository.AddAsync(new HeartRateRecord
        {
            Timestamp = DateTime.Now,
            HeartRate = 75,
            DeviceId = "test-device"
        });
        
        // Act
        await _repository.DeleteBeforeDateAsync(DateTime.Now.AddDays(-30));
        
        // Assert
        var remaining = await _repository.GetByTimeRangeAsync(
            DateTime.Now.AddDays(-32), DateTime.Now);
        Assert.AreEqual(1, remaining.Count());
    }
}
```

```csharp
[TestClass]
[TestCategory("Performance")]
public class MemoryUsageTests
{
    [TestMethod]
    public void HeartRateDataBuffer_StaysWithinMemoryLimit()
    {
        // Arrange
        var buffer = new HeartRateDataBuffer(300);
        var initialMemory = GC.GetTotalMemory(true);
        
        // Act
        for (int i = 0; i < 10000; i++)
        {
            buffer.Add(new HeartRateData
            {
                HeartRate = 72,
                Timestamp = DateTime.Now,
                DeviceId = "test"
            });
        }
        
        var finalMemory = GC.GetTotalMemory(true);
        var memoryIncrease = finalMemory - initialMemory;
        
        // Assert - 应该小于1MB
        Assert.IsTrue(memoryIncrease < 1024 * 1024);
    }
    
    [TestMethod]
    public void LongRunning_NoMemoryLeak()
    {
        // Arrange
        var initialMemory = GC.GetTotalMemory(true);
        var service = new HeartRateService();
        
        // Act - 模拟长时间运行
        for (int i = 0; i < 100000; i++)
        {
            service.ProcessHeartRateData(new HeartRateData
            {
                HeartRate = 72,
                Timestamp = DateTime.Now,
                DeviceId = "test"
            });
        }
        
        GC.Collect();
        GC.WaitForPendingFinalizers();
        var finalMemory = GC.GetTotalMemory(true);
        
        // Assert - 内存增长应该很小
        Assert.IsTrue(finalMemory - initialMemory < 10 * 1024 * 1024);
    }
}
```
